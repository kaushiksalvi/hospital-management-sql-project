Create Database HospitalManagementSystem;
Use HospitalManagementSystem;

Create Table Department 
(
	dept_Id Int Primary Key,
	dept_Name Varchar(100)
);

Create Table Room 
(
	room_No Int Primary Key,
	dept_Id Int,
	room_Type Varchar(100),
	Foreign Key (dept_Id) References Department(dept_Id)
);

Create Table Doctor 
(
	doct_Id Int Primary Key,
	dept_Id Int,
	FName Varchar(100),
	LName Varchar(100),
	Gender CHAR,
	contact_No Varchar(100),
	surgeon_Type Varchar(100),
	office_No Int,
	Foreign Key (dept_Id) References Department(dept_Id),
	Foreign Key (office_No) References Room(room_No)
);

Create Table Nurse 
(
	nurse_Id Int Primary Key,
	dept_Id Int,
	FName Varchar(100),
	LName Varchar(100),
	Gender Char,
	conatct_No Varchar(100),
	Foreign Key (dept_Id) References Department(dept_Id)
);

Create Table Helpers 
(
	helper_Id Int Primary Key,
	dept_Id Int,
	FName Varchar(100),
	LName Varchar(100),
	Gender Char,
	contact_No Varchar(100),
	Foreign Key (dept_Id) References Department(dept_Id)
);

Create Table Ward 
(
	ward_No Int Primary Key,
	ward_Name Varchar(100),
	dept_Id Int,
	Foreign Key (dept_Id) References Department(dept_Id)
);

Create Table Bed 
(
	bed_No Int Primary Key,
	ward_No Int,
	Foreign Key (ward_No) References Ward(ward_No)
);

Create Table Patients 
(
	patient_Id Int Primary Key,
	FName Varchar(100),
	LName Varchar(100),
	Gender Char,
	Date_Of_Birth Date,
	contact_No Varchar(100),
	pt_Address Varchar(100)
);

Create Table BedRecords 
(
	admission_Id Int Primary Key,
	bed_No Int,
	patient_Id Int,
	nurse_Id Int,
	helper_Id Int,
	admission_Date Date,
	discharge_Date Date,
	amount Int,
	mode_of_payment Varchar(50),
	Foreign Key (bed_No) References Bed(bed_No),
	Foreign Key (patient_Id) References Patients(patient_Id),
	Foreign Key (nurse_Id) References Nurse(nurse_Id),
	Foreign Key (helper_Id) References Helpers(helper_Id)
);

Create Table RoomRecords 
(
	admisson_ID Int Primary Key,
	room_no Int,
	patient_Id Int,
	nurse_Id Int,
	helper_Id Int,
	admission_Date Date,
	discharge_Date Date,
	amount Int,
	mode_of_payment Varchar(50),
	Foreign Key (room_no) References Room(room_No),
	Foreign Key (patient_Id) References Patients(patient_Id),
	Foreign Key (nurse_Id) References Nurse(nurse_Id),
	Foreign Key (helper_Id) References Helpers(helper_Id)
);

Create Table Appointment 
(
	appoIntment_Id Int Primary Key,
	patient_Id Int,
	doct_Id Int,
	reason Varchar(100),
	appointment_Date Date,
	payment_amount Int,
	mode_of_payment Varchar(100),
	mode_of_appointment Varchar(100),
	appointment_status Varchar(100),
	Foreign Key (patient_Id) References Patients(patient_Id),
	Foreign Key (doct_Id) References Doctor(doct_Id)
);

Create Table MedicalRecord 
(
	record_Id Int Primary Key,
	doct_Id Int,
	patient_Id Int,
	visit_Date Date,
	curr_Weight Decimal(10,2),
	curr_height Decimal(10,2),
	curr_Blood_Pressure Varchar(100),
	curr_Temp_F Decimal(10,2),
	diagnosis Varchar(500),
	treatment Varchar(100),
	next_Visit Date,
	Foreign Key (doct_Id) References Doctor(doct_Id),
	Foreign Key (patient_Id) References Patients(patient_Id)
);

Create Table StaffShift 
(
	shift_Id Int Primary Key,
	doct_Id Int,
	nurse_Id Int,
	helper_Id Int,
	shift_Date Date,
	shift_Start Time,
	shift_End Time,
	Foreign Key (doct_Id) References Doctor(doct_Id),
	Foreign Key (nurse_Id) References Nurse(nurse_Id),
	Foreign Key (helper_Id) References Helpers(helper_Id)
);

Create Table SurgeryRecord 
(
	surgery_Id Int Primary Key,
	patient_Id Int,
	surgeon_Id Int,
	surgery_Type Varchar(100),
	surgery_Date Date,
	start_Time Time,
	end_Time Time,
	room_no Int,
	notes Varchar(1000),
	nurse_Id Int,
	helper_Id Int,
	Foreign Key (patient_Id) References Patients(patient_Id),
	Foreign Key (surgeon_Id) References Doctor(doct_Id),
	Foreign Key (room_no) References Room(room_No),
	Foreign Key (nurse_Id) References Nurse(nurse_Id),
	Foreign Key (helper_Id) References Helpers(helper_Id)
);

select d.* from doctor d
join department de on de.dept_id = d.dept_id
where dept_name = "Cardiology" ;

# Q3 List all appointments scheduled on a specific date.

select * from appointment

# Q4 Find the contact number of a patient by patient name.

# Q5 Display all departments in alphabetical order.

select * from department

# Q6 Display patient name, doctor name, and appointment date.

select p.Fname,d.Fname,appointment_date from doctor d 
join appointment a on a.doct_Id = d.doct_Id
join patients p on p.patient_Id = a.patient_Id;

# Q7 List doctors along with their department names.

select d.Fname  as doctor_name, dep.dept_Name from doctor d 
join department dep on dep.dept_Id = d.dept_Id;

# Q8 Show all patients who have taken appointments with a Neurology doctor.

SELECT DISTINCT p.FName
FROM Patients p
JOIN Appointment a ON p.patient_Id = a.patient_Id
JOIN Doctor d ON a.doct_Id = d.doct_Id
JOIN Department dep ON d.dept_Id = dep.dept_Id
WHERE dep.dept_Name = 'Neurology';


# Q9 Display patients who have never taken any appointment.

select p.FName from patients p 
left join appointment a on a.patient_Id = p.patient_Id

# Q11 Count total number of patients.

select count(*) as total_patient from patients;

# Q12 Find total appointments handled by each doctor.

select d.FName, sum(a.appointment_Id) as total_aapointments  from appointment a 
right join doctor d on d.doct_Id = a.doct_Id 
group by d.FName;


# Q13 Display the department-wise doctor count.
select count(d.doct_Id) as total_doctors,dep.dept_Name from doctor d 
join department dep on dep.dept_Id = d.dept_id
group by dep.dept_Name ;


# Q14 Find the total revenue generated from appointments.

select sum(payment_amount) as total_revenue from appointment;

# Q15 Show average appointment fee per department.

select avg(a.payment_amount) as avgrage_fees, dep.dept_Name from doctor d 
join appointment a on a.doct_Id = d.doct_Id 
join department dep on dep.dept_id = d.dept_Id
group by dep.dept_Name;


# Q16 Display doctors who have handled more than 5 appointments.

select d.FName,d.doct_Id, count(a.appointment_id) from doctor d
join appointment a on a.doct_Id = d.doct_Id 
group by d.FName,d.doct_Id
having count(a.appointment_id) > 5;

# Q17 Find departments having more than 3 doctors.

select dep.dept_Name, count(d.doct_id)  from department dep 
join doctor d on d.dept_Id = dep.dept_Id 
group by dep.dept_Name
having count(d.doct_Id) > 3;

# Q18 Show doctors whose total appointment revenue is above 2000.

select d.FName, sum(a.payment_amount) from doctor d 
join appointment a on a.doct_Id = d.doct_Id 
group by d.FName
having sum(payment_amount) > 2000;

# Q19 Find departments with zero appointments.

select dep.dept_name from department dep
left join doctor d on d.dept_Id = dep.dept_Id
left join appointment a on a.doct_Id = d.doct_Id

# Q20 List patients who visited the hospital more than once.

select p.FName , count(a.appointment_Id) from patients p 
join appointment a on a.patient_Id = p.patient_Id 
group by p.FName 
having count(a.appointment_Id )> 1;

# Q22 Display the doctor who handled the maximum number of patients.

SELECT *
FROM Doctor
WHERE doct_Id in (
    SELECT doct_Id
    FROM Appointment
    GROUP BY doct_Id
    HAVING COUNT(DISTINCT patient_Id) = (
        SELECT MAX(cnt)
        FROM (
            SELECT COUNT(DISTINCT patient_Id) AS cnt
            FROM Appointment
            GROUP BY doct_Id
        ) t
    )
);

#Q 23 Show departments whose doctor count is above average.

SELECT *
FROM Department
WHERE dept_Id IN (
    SELECT dept_Id
    FROM Doctor
    GROUP BY dept_Id
    HAVING COUNT(*) >
    (
        SELECT AVG(dept_count)
        FROM (
            SELECT COUNT(*) AS dept_count
            FROM Doctor
            GROUP BY dept_Id
        ) x
    )
);

 
# Q24 Find patients who visited the same doctor more than once.

SELECT *
FROM Patients
WHERE patient_Id IN (
    SELECT patient_Id
    FROM Appointment
    GROUP BY patient_Id, doct_Id
    HAVING COUNT(*) > 1
);

# Q25 Display appointments taken by the youngest patient.

SELECT *
FROM Appointment
WHERE patient_Id = (
    SELECT patient_Id
    FROM Patients
    WHERE Date_Of_Birth = (
        SELECT MAX(Date_Of_Birth)
        FROM Patients
    )
);
